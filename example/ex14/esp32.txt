#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C lcd(0x27, 16, 2);
void setup() {
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0);     // ไปที่ตัวอักษรที่ 0 แถวที่ 1
  lcd.print("Welcome To"); 
  lcd.setCursor(6, 1);     // ไปที่ตัวอักษรที่ 6 แถวที่ 2
  lcd.print("My ESP32");
}
void loop() {

}
