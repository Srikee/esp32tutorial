#include <ESP32Servo.h>
Servo myservo; //ประกาศตัวแปรแทน Servo
void setup() {
  myservo.attach(4); // กำหนดขา 4 ควบคุม Servo
}
void loop() {
  myservo.write(0); // หมุนไปองศาที่ 0
  delay(1000); // หน่วงเวลา 1000ms
  myservo.write(90); // หมุนไปองศาที่ 90
  delay(1000); // หน่วงเวลา 1000ms
  myservo.write(180); // หมุนไปองศาที่ 180
  delay(1000); // หน่วงเวลา 1000ms
}
